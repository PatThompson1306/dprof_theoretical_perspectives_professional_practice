\begin{description}[
  leftmargin=1.4cm,
  labelwidth=1.2cm,
  labelsep=0.2cm,
  itemsep=4pt,
  parsep=0pt,
  font=\normalfont\bfseries\small
]
  \item[EC1] Sources published prior to 2018 unless meeting the threshold defined
    in IC7; all exceptions recorded in the review log.
  \item[EC2] Non-peer-reviewed opinion pieces, blog posts, or media articles without
    evidential grounding, unless produced by authoritative professional or statutory
    bodies.
  \item[EC3] Studies focused solely on adult sexual offending with no transferable
    applicability or analytic relevance to child safeguarding contexts.
  \item[EC4] Technical \gls{nlp} literature without contextual application to safeguarding,
    criminal justice, or operational deployment scenarios.
  \item[EC5] Technology adoption studies from sectors lacking structural or cultural
    comparability to UK policing (e.g.\ private sector commercial contexts without
    analogous accountability frameworks).
  \item[EC6] Duplicate publications; where multiple versions of the same study exist,
    the most recent peer-reviewed version is retained and earlier versions excluded.
  \item[EC7] Sources for which full text cannot be obtained through institutional
    access, inter-library loan, or open-access repositories within a reasonable
    retrieval window.
\end{description}
\end{tcolorbox}
