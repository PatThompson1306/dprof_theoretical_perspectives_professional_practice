\begin{description}[

  \item[IC1] Peer-reviewed journal articles, conference proceedings, and edited book
    chapters published 2018 to present.
  \item[IC2] Grey literature from authoritative sources (\gls{npcc}, Home Office, \gls{ceop},
    \gls{nist}, \gls{ico}) published 2018 to present.
  \item[IC3] Studies addressing \gls{nlp}, machine learning, or computational linguistics
    applied to safeguarding, criminal justice, or analogous operational contexts.
  \item[IC4] Literature examining grooming, \gls{csa}, or online predatory behaviour with
    linguistic, behavioural, or empirical focus.
  \item[IC5] Organisational, cultural, or governance studies relating to technology
    adoption within policing or structurally comparable public sector contexts.
  \item[IC6] English-language sources; non-English sources permitted where a
    peer-reviewed translation or substantive abstract provides sufficient evidential
    grounding with explicit methodological justification.
  \item[IC7] Pre-2018 foundational works where theoretical significance is established
    and no substantively updated equivalent exists, subject to explicit justification
    in the synthesis narrative.
\end{description}
