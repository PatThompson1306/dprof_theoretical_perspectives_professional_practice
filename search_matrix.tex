\textcolor{track1}{\textbf{Criminology \& Behavioural Science}} &

\textit{What linguistic and behavioural precursors of contact sex offending exist in
offender--victim conversations, and how do grooming typologies characterise
the progression toward contact abuse?} &
%
\begin{itemize}[leftmargin=1em,nosep,topsep=2pt,partopsep=0pt]
  \item Scopus
  \item PsycINFO
  \item Web of Science
  \item Criminal Justice Abstracts
  \item NCJRS
  \item CINAHL
  \item ASSIA
  \item LLBA
  \item Google Scholar
\end{itemize} &
%
\begin{itemize}[leftmargin=1em,nosep,topsep=2pt,partopsep=0pt]
  \item \textbf{Scopus} --- broad interdisciplinary coverage with strong indexing of criminology and psychology journals.
  \item \textbf{PsycINFO} --- essential for the psychological and behavioural dimensions of offending, including victimology and typology research.
  \item \textbf{Web of Science} --- high-quality peer-reviewed social-science journals relevant to behavioural precursors.
  \item \textbf{Criminal Justice Abstracts} (via EBSCOhost) --- focused criminological literature with strong coverage of sex-offending typologies.
  \item \textbf{Google Scholar} --- grey literature, NGO reports, and regulatory publications alongside mainstream citation trails.
\end{itemize} \\[8pt]

  \item Papers With Code
  \item LREC\nobreakdash-COLING Anthology

  \item \textbf{Scopus} --- interdisciplinary bridge between technical and applied criminological literature.
