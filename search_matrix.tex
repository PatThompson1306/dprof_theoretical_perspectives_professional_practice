\textcolor{track2}{\textbf{Computer Science \& \gls{nlp}}} &
%
\textit{Which \gls{nlp} methods can be most effectively realised within policing
infrastructure, and what does the evidence base reveal about model performance,
explainability, and operational deployment constraints?} &
%
\begin{itemize}[leftmargin=*,nosep,topsep=2pt,partopsep=0pt,itemindent=0pt,align=left]
  \item IEEE Xplore
  \item ACM Digital Library
  \item ACL Anthology
  \item arXiv (cs.CL, cs.AI, cs.CY)
  \item Papers With Code
  \item Scopus
\end{itemize} &
%
\begin{itemize}[leftmargin=*,nosep,topsep=2pt,partopsep=0pt,itemindent=0pt,align=left]
  \item \textbf{IEEE Xplore} --- \gls{ieee} curated database for applied \gls{nlp} architecture, development and deployment.
  \item \textbf{ACM Digital Library} --- Curated by the \gls{acm}; similar in stature for \gls{nlp} content as \textbf{IEEE Xplore}.
  \item \textbf{ACL Anthology} --- Curated by the \gls{acl}; linguistics-grounded \gls{nlp} resource with a focus on language understanding and generation, essential for conversational analysis.
  \item \textbf{arXiv} (cs.CL) --- Pre-print access to novel language model research.
  \item \textbf{Papers With Code} --- Community driven repository linking code implementations to research papers, crucial for identifying reproducible \gls{nlp} methods and benchmarks.
  \item \textbf{Scopus} --- Interdisciplinary bridge between technical and applied criminological literature.
\end{itemize} \\[8pt]
